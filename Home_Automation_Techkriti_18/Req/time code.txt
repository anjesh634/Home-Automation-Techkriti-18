#include <LiquidCrystal.h>
LiquidCrystal
lcd(12,11,5,4,3,2);
int h=12;
int m=0;
int s=0;
void setup()
{
  lcd.begin(16,2);
   lcd.setCursor(6,0);
  lcd.print("TIME");
  lcd.setCursor(4,1);
  lcd.print(h);
  lcd.print(":");
  lcd.print(m);
  lcd.print(m);
  lcd.print(":");
  lcd.print(s);
  lcd.print(s);
}
void loop()
{
 if(s<10)
 {  
    delay(80);
    lcd.setCursor(11,1);
    lcd.print(s);
 } 
  else{
    delay(80);
    lcd.setCursor(10,1);
    lcd.print(s);
  }
    s=s+1;
  if(s==60)
  {
    s=0;
    lcd.setCursor(10,1);
    lcd.print(s);
    lcd.print(s);
    if(m<10)
    {
      m=m+1;
     lcd.setCursor(8,1);
     lcd.print(m);
    }
    else{
       m=m+1;
     lcd.setCursor(7,1);
     lcd.print(m);
    }
  }
    if(m==59)
  {
    m=0;
    lcd.setCursor(7,1);
    lcd.print(h);
    if(h<10)
    {
      h=h+1;
     lcd.setCursor(5,1);
     lcd.print(h);
    }
    else{
       h=h+1;
     lcd.setCursor(4,1);
     lcd.print(h);
    }
  }
  if(h>12)
  {
    h=1;
  }
}
